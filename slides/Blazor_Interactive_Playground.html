@page "/tareas" <input @bind="nuevaTarea" /> <button @onclick="AgregarTarea">Agregar</button> <!-- Loop con @foreach --> @foreach (var tarea in tareas) { <div> @tarea <button @onclick="() => Eliminar(tarea)"> 🗑️ </button> </div> } @code { private string nuevaTarea = ""; private List<string> tareas = new(); private void AgregarTarea() { if (!string.IsNullOrWhiteSpace(nuevaTarea)) { tareas.Add(nuevaTarea); nuevaTarea = ""; } } private void Eliminar(string tarea) { tareas.Remove(tarea); } }
💡 @foreach: Itera sobre una lista y genera HTML para cada elemento. Como un for loop pero en el HTML.

@page "/registro" <h3>Formulario de Registro</h3> <input @bind="nombre" placeholder="Nombre" /> <input @bind="edad" type="number" /> <input @bind="email" type="email" /> <button @onclick="Registrar">Registrar</button> @if (registroExitoso) { <div class="alert alert-success"> ✅ Registro exitoso: @nombre </div> } @if (registros.Count > 0) { <h4>Usuarios Registrados:</h4> @foreach (var reg in registros) { <div>@reg</div> } } @code { private string nombre = ""; private int edad = 0; private string email = ""; private bool registroExitoso = false; private List<string> registros = new(); private void Registrar() { if (!string.IsNullOrWhiteSpace(nombre)) { registros.Add($"{nombre} - {edad} años"); registroExitoso = true; nombre = ""; edad = 0; email = ""; } } }
